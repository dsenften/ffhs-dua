== Ihr erstes Programm

=== Stack-Beispiel

[source,python]
----
from src.algs4.fundamentals.stack import Stack

# Stack erstellen
stack = Stack[str]()

# Elemente hinzufügen
stack.push("Erstes Element")
stack.push("Zweites Element")

# Element entfernen
element = stack.pop()
print(f"Entfernt: {element}")  # "Zweites Element"
----

=== Test schreiben

[source,python]
----
# tests/test_mein_stack.py
from src.algs4.fundamentals.stack import Stack

def test_mein_stack():
    stack = Stack[int]()
    stack.push(42)
    assert stack.pop() == 42
    assert stack.is_empty()
----

=== Test ausführen

[source,bash]
----
uv run pytest tests/test_mein_stack.py -v
----
